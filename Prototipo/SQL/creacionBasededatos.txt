CREATE TABLE PERSONA(
   PERSONA_NOMBRE varchar NOT NULL,
   PERSONA_TELEFONO varchar NOT NULL,
   PERSONA_CEDULA varchar NOT NULL,
   PRIMARY KEY( PERSONA_CEDULA )
);

CREATE TABLE FACTURA(
   FACTURA_FECHA date NOT NULL,
   FACTURA_DESCRIPCION varchar NOT NULL,
   FACTURA_MARCA varchar NOT NULL,
   FACTURA_VALOR int NOT NULL,
FACTURA_CLAVE int NOT NULL,
   FACTURA_ID BIGSERIAL,
   FACTURA_CEDULA varchar REFERENCES PERSONA(PERSONA_CEDULA) NOT NULL,
   PRIMARY KEY(FACTURA_ID)
);
